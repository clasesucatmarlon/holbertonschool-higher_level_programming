The 1-My list module
======================
import file:

>>> MyList = __import__('1-l').MyList


test empty list
-----------------------------
>>> l = MyList()
>>> l.print_sorted()
[]



test three items
-----------------------------
>>> l.append(3)
>>> l.append(21)
>>> l.append(4)
>>> print(l)
[3, 21, 4]

>>> l.print_sorted()
[3, 4, 21]


test negative items
-----------------------------
>>> l = MyList()
>>> l.append(-3)
>>> l.append(-21)
>>> l.append(-4)
>>> l.print_sorted()
[-21, -4, -3]


test negative and positives items
---------------------------------------------
>>> l = MyList()
>>> l.append(3)
>>> l.append(-21)
>>> l.append(-4)
>>> l.append(-3)
>>> l.append(21)
>>> l.append(4)
>>> l.append(-7)
>>> l.print_sorted()
